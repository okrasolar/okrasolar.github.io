102 virtual Size pull(T* data, Size size)
103 {
104 if (size > bytesAvailable())
105 size = bytesAvailable();
106
107 Size readCount = 0;
108
109 while (readCount < size) {
110 data[readCount++] = _buffer[_readHead];
111 incrementOrResetHead(_readHead);
112 }
113 _availableData -= readCount;
114
115 return readCount;
